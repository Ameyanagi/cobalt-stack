1//! Email delivery service for user communication.
2//!
3//! This module provides email sending functionality with support for both
4//! mock (development) and production SMTP implementations. Currently handles
5//! email verification but designed to be extensible for other email types.
6//!
7//! # Architecture
8//!
9//! - **EmailSender trait**: Abstraction for different email backends
10//! - **MockEmailSender**: Development implementation that logs to console
11//! - **verification**: Email verification token management
12//!
13//! # Usage
14//!
15//! ```no_run
16//! use cobalt_stack_backend::services::email::{EmailSender, MockEmailSender};
17//!
18//! let sender = MockEmailSender;
19//! sender.send_verification_email("user@example.com", "abc123token").unwrap();
20//! ```
21//!
22//! # Future Extensions
23//!
24//! - SMTP implementation for production email delivery
25//! - Template-based email rendering
26//! - Password reset emails
27//! - Welcome emails
28//! - Notification emails

35/// Abstraction for email sending implementations.
36///
37/// This trait allows swapping between mock (development) and real (production)
38/// email backends without changing handler code. Implementations handle the
39/// actual delivery mechanism.
40///
41/// # Implementations
42///
43/// - [`MockEmailSender`]: Logs to console instead of sending real emails
44/// - Future: `SmtpEmailSender` for production SMTP delivery
45///
46/// # Examples
47///
48/// ```
49/// use cobalt_stack_backend::services::email::{EmailSender, MockEmailSender};
50///
51/// fn send_verification(sender: &impl EmailSender, email: &str, token: &str) {
52///     sender.send_verification_email(email, token).unwrap();
53/// }
54///
55/// let mock_sender = MockEmailSender;
56/// send_verification(&mock_sender, "user@example.com", "abc123");
57/// ```
58pub trait EmailSender {
59    /// Send an email verification link to the user.
60    ///
61    /// # Arguments
62    ///
63    /// * `to` - Recipient email address
64    /// * `token` - Verification token to include in link
65    ///
66    /// # Returns
67    ///
68    /// - `Ok(())` - Email sent successfully (or logged for mock)
69    /// - `Err(_)` - Email delivery failed
70    fn send_verification_email(&self, to: &str, token: &str) -> Result<()>;
71}
