8/// Store a refresh token in the database
9///
10/// The token is hashed before storage for security
11pub async fn store_refresh_token(
12    db: &DatabaseConnection,
13    user_id: Uuid,
14    token: &str,
15    jti: Uuid,
16    expires_in_days: i64,
17) -> Result<()> {
18    let token_hash = hash_token(token);
19    let expires_at = Utc::now() + Duration::days(expires_in_days);
20
21    let refresh_token = refresh_tokens::ActiveModel {
22        id: Set(jti),
23        user_id: Set(user_id),
24        token_hash: Set(token_hash),
25        expires_at: Set(expires_at.into()),
26        revoked_at: Set(None),
27        created_at: Set(Utc::now().into()),
28    };
29
30    refresh_token.insert(db).await?;
31    Ok(())
32}

34/// Validate a refresh token
35///
36/// Checks that:
37/// - Token exists in database
38/// - Token hash matches
39/// - Token is not expired
40/// - Token is not revoked
41pub async fn validate_refresh_token(
42    db: &DatabaseConnection,
43    token: &str,
44    jti: Uuid,
45) -> Result<Uuid> {
46    let token_hash = hash_token(token);
47
48    let stored_token = RefreshTokens::find_by_id(jti)
49        .one(db)
50        .await?
51        .ok_or_else(|| AuthError::InvalidToken)?;
52
53    // Check if token hash matches
54    if stored_token.token_hash != token_hash {
55        return Err(AuthError::InvalidToken.into());
56    }
57
58    // Check if token is revoked
59    if stored_token.revoked_at.is_some() {
60        return Err(AuthError::TokenBlacklisted.into());
61    }
62
63    // Check if token is expired
64    let now: DateTime<Utc> = Utc::now();
65    let expires_at: DateTime<Utc> = stored_token.expires_at.with_timezone(&Utc);
66    if expires_at < now {
67        return Err(AuthError::TokenExpired.into());
68    }
69
70    Ok(stored_token.user_id)
71}

73/// Revoke a refresh token
74///
75/// Marks the token as revoked in the database
76pub async fn revoke_refresh_token(db: &DatabaseConnection, jti: Uuid) -> Result<()> {
77    let stored_token = RefreshTokens::find_by_id(jti)
78        .one(db)
79        .await?
80        .ok_or_else(|| AuthError::InvalidToken)?;
81
82    let mut active_token: refresh_tokens::ActiveModel = stored_token.into();
83    active_token.revoked_at = Set(Some(Utc::now().into()));
84    active_token.update(db).await?;
85
86    Ok(())
87}

89/// Rotate refresh token (revoke old, issue new)
90///
91/// This implements token rotation pattern for enhanced security
92pub async fn rotate_refresh_token(
93    db: &DatabaseConnection,
94    old_jti: Uuid,
95    new_token: &str,
96    new_jti: Uuid,
97    user_id: Uuid,
98    expires_in_days: i64,
99) -> Result<()> {
100    // Revoke old token
101    revoke_refresh_token(db, old_jti).await?;
102
103    // Store new token
104    store_refresh_token(db, user_id, new_token, new_jti, expires_in_days).await?;
105
106    Ok(())
107}

109/// Revoke all refresh tokens for a user (logout from all devices)
110pub async fn revoke_all_user_tokens(db: &DatabaseConnection, user_id: Uuid) -> Result<()> {
111    let tokens = RefreshTokens::find()
112        .filter(refresh_tokens::Column::UserId.eq(user_id))
113        .filter(refresh_tokens::Column::RevokedAt.is_null())
114        .all(db)
115        .await?;
116
117    for token in tokens {
118        let mut active_token: refresh_tokens::ActiveModel = token.into();
119        active_token.revoked_at = Set(Some(Utc::now().into()));
120        active_token.update(db).await?;
121    }
122
123    Ok(())
124}

126/// Clean up expired tokens (for maintenance tasks)
127///
128/// Deletes tokens that have been expired for more than retention_days
129pub async fn cleanup_expired_tokens(
130    db: &DatabaseConnection,
131    retention_days: i64,
132) -> Result<u64> {
133    let cutoff = Utc::now() - Duration::days(retention_days);
134
135    let result = RefreshTokens::delete_many()
136        .filter(refresh_tokens::Column::ExpiresAt.lt(cutoff))
137        .exec(db)
138        .await?;
139
140    Ok(result.rows_affected)
141}

143/// Hash a token for storage
144///
145/// Uses SHA-256 for token hashing
146fn hash_token(token: &str) -> String {
147    let mut hasher = Sha256::new();
148    hasher.update(token.as_bytes());
149    format!("{:x}", hasher.finalize())
150}
