1//! Token blacklist service for JWT revocation.
2//!
3//! This module provides functionality to blacklist JWT access tokens, enabling
4//! immediate token revocation for logout, security incidents, or administrative
5//! actions. Uses Valkey/Redis with automatic TTL expiry.
6//!
7//! # Architecture
8//!
9//! - **Key Format**: `blacklist:{token}` stored with value `1`
10//! - **TTL Management**: Tokens automatically expire when they would naturally expire
11//! - **Fast Lookup**: O(1) Redis GET operation for blacklist checks
12//! - **Memory Efficient**: Expired entries automatically removed by Redis
13//!
14//! # Use Cases
15//!
16//! - **User Logout**: Immediately invalidate access token
17//! - **Security Incidents**: Revoke compromised tokens
18//! - **Account Suspension**: Block all user tokens
19//! - **Password Change**: Optionally invalidate existing sessions
20//!
21//! # Security Notes
22//!
23//! - Only blacklist access tokens, not refresh tokens (use database for refresh tokens)
24//! - TTL must match access token expiry to prevent premature removal
25//! - Blacklist checks add small latency to protected endpoints
26//! - Consider connection pooling for high-traffic applications
27//!
28//! # Examples
29//!
30//! ```no_run
31//! use cobalt_stack_backend::services::valkey::blacklist::{add_to_blacklist, is_blacklisted};
32//! use redis::Client;
33//!
34//! # fn example() -> anyhow::Result<()> {
35//! let client = Client::open("redis://127.0.0.1/")?;
36//! let mut conn = client.get_connection()?;
37//!
38//! // Blacklist token for 30 minutes (1800 seconds)
39//! add_to_blacklist(&mut conn, "expired_token_123", 1800)?;
40//!
41//! // Check if token is blacklisted
42//! assert!(is_blacklisted(&mut conn, "expired_token_123")?);
43//! assert!(!is_blacklisted(&mut conn, "valid_token_456")?);
44//! # Ok(())
45//! # }
46//! ```

51/// Add a JWT access token to the blacklist with automatic expiry.
52///
53/// Stores the token in Valkey/Redis with a TTL matching the token's remaining
54/// lifetime. Once added, the token will fail all authentication checks until
55/// it naturally expires.
56///
57/// # Arguments
58///
59/// * `conn` - Active Valkey/Redis connection
60/// * `token` - JWT access token to blacklist (full token string)
61/// * `ttl` - Time to live in seconds (should match token's exp - now)
62///
63/// # Returns
64///
65/// - `Ok(())` - Token successfully added to blacklist
66/// - `Err(_)` - Redis connection or command error
67///
68/// # Examples
69///
70/// ```no_run
71/// use cobalt_stack_backend::services::valkey::blacklist::add_to_blacklist;
72/// use redis::Client;
73///
74/// # fn example() -> anyhow::Result<()> {
75/// let client = Client::open("redis://127.0.0.1/")?;
76/// let mut conn = client.get_connection()?;
77///
78/// // Blacklist token that expires in 30 minutes
79/// let token = "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...";
80/// add_to_blacklist(&mut conn, token, 1800)?;
81/// # Ok(())
82/// # }
83/// ```
84///
85/// # Security Notes
86///
87/// - TTL should be calculated from token's actual expiry time
88/// - Setting TTL too short allows token to work after removal from blacklist
89/// - Setting TTL too long wastes Redis memory unnecessarily
90/// - Use this for access tokens only (refresh tokens use database revocation)
91pub fn add_to_blacklist(conn: &mut Connection, token: &str, ttl: i64) -> Result<()> {
92    let key = format!("blacklist:{}", token);
93    conn.set_ex::<_, _, ()>(&key, 1, ttl as u64)?;
94    Ok(())
95}

97/// Check if a JWT access token has been blacklisted.
98///
99/// Performs a fast O(1) Redis EXISTS check to determine if the token
100/// has been revoked. Should be called during authentication to reject
101/// blacklisted tokens.
102///
103/// # Arguments
104///
105/// * `conn` - Active Valkey/Redis connection
106/// * `token` - JWT access token to check (full token string)
107///
108/// # Returns
109///
110/// - `Ok(true)` - Token is blacklisted and should be rejected
111/// - `Ok(false)` - Token is not blacklisted (may still be invalid for other reasons)
112/// - `Err(_)` - Redis connection or command error
113///
114/// # Examples
115///
116/// ```no_run
117/// use cobalt_stack_backend::services::valkey::blacklist::{add_to_blacklist, is_blacklisted};
118/// use redis::Client;
119///
120/// # fn example() -> anyhow::Result<()> {
121/// let client = Client::open("redis://127.0.0.1/")?;
122/// let mut conn = client.get_connection()?;
123///
124/// let token = "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...";
125///
126/// // Initially not blacklisted
127/// assert!(!is_blacklisted(&mut conn, token)?);
128///
129/// // After blacklisting
130/// add_to_blacklist(&mut conn, token, 1800)?;
131/// assert!(is_blacklisted(&mut conn, token)?);
132/// # Ok(())
133/// # }
134/// ```
135///
136/// # Performance
137///
138/// - O(1) time complexity (Redis EXISTS command)
139/// - Typical latency: <1ms on local Redis
140/// - Consider connection pooling for concurrent requests
141///
142/// # Error Handling
143///
144/// If Redis is unavailable, this function returns an error. Depending on your
145/// security requirements, you may want to:
146/// - Fail secure: reject all requests if blacklist check fails
147/// - Fail open: allow requests if blacklist check fails (risky)
148pub fn is_blacklisted(conn: &mut Connection, token: &str) -> Result<bool> {
149    let key = format!("blacklist:{}", token);
150    let exists: bool = conn.exists(&key)?;
151    Ok(exists)
152}
