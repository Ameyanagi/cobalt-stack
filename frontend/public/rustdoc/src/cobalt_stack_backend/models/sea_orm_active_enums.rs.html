1//! SeaORM active enums for database enum types.
2//!
3//! This module defines Rust enums that map to PostgreSQL ENUM types.
4//! These enums are used in entity fields to provide type-safe access
5//! to database enum values.
6
7use sea_orm::entity::prelude::*;
8use serde::{Deserialize, Serialize};
9use utoipa::ToSchema;
10
11/// User role enum for role-based access control (RBAC).
12///
13/// Defines the authorization level for a user account.
14/// Used in middleware to enforce access control policies.
15///
16/// # Variants
17///
18/// - `User`: Standard user with access to their own resources
19/// - `Admin`: Administrator with elevated privileges
20///
21/// # Database Mapping
22///
23/// Maps to PostgreSQL ENUM `user_role` with values:
24/// - `"user"` → [`UserRole::User`]
25/// - `"admin"` → [`UserRole::Admin`]
26///
27/// # Examples
28///
29/// ```
30/// use cobalt_stack::models::sea_orm_active_enums::UserRole;
31///
32/// let role = UserRole::User;
33/// assert_eq!(serde_json::to_string(&role).unwrap(), "\"user\"");
34///
35/// let admin = UserRole::Admin;
36/// assert_eq!(serde_json::to_string(&admin).unwrap(), "\"admin\"");
37/// ```
38#[derive(Debug, Clone, PartialEq, Eq, EnumIter, DeriveActiveEnum, Serialize, Deserialize, ToSchema)]
39#[sea_orm(rs_type = "String", db_type = "Enum", enum_name = "user_role")]
40#[serde(rename_all = "lowercase")]
41pub enum UserRole {
42    /// Standard user role with basic permissions.
43    ///
44    /// Users can:
45    /// - Access their own profile
46    /// - Update their own settings
47    /// - View public resources
48    #[sea_orm(string_value = "user")]
49    User,
50
51    /// Administrator role with elevated permissions.
52    ///
53    /// Admins can:
54    /// - Access all user accounts
55    /// - Disable/enable user accounts
56    /// - View system-wide statistics
57    /// - Perform administrative operations
58    #[sea_orm(string_value = "admin")]
59    Admin,
60}
